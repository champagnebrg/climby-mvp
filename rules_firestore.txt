rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    function verifiedEmail() {
      return signedIn() && request.auth.token.email_verified == true;
    }

    function userDoc() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid));
    }

    function isSuperadmin() {
      return signedIn()
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && userDoc().data.role == 'superadmin';
    }

    function isGymAdminRole() {
      return signedIn()
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && (userDoc().data.role == 'gym_admin' || userDoc().data.role == 'admin');
    }

    function managesGym(gymId) {
      return isGymAdminRole() && userDoc().data.gymManaged == gymId;
    }

    function canManageGym(gymId) {
      return isSuperadmin() || isGymOwner(gymId) || managesGym(gymId);
    }

    function isSelf(uid) {
      return signedIn() && request.auth.uid == uid;
    }

    function isGymOwner(gymId) {
      return signedIn()
        && exists(/databases/$(database)/documents/gyms/$(gymId))
        && get(/databases/$(database)/documents/gyms/$(gymId)).data.owner == request.auth.uid;
    }

    match /users/{uid} {
      allow read: if verifiedEmail() || isSuperadmin();
      allow create: if isSuperadmin() || (
        isSelf(uid)
        && request.resource.data.authUid == request.auth.uid
      );
      allow update, delete: if isSuperadmin() || (
        isSelf(uid)
        && resource.data.authUid == request.auth.uid
      );

      match /routeProgress/{progressId} {
        allow read: if verifiedEmail() || isSuperadmin();
        allow create, update: if isSuperadmin() || (verifiedEmail() && isSelf(uid));
        allow delete: if isSuperadmin() || (verifiedEmail() && isSelf(uid)) || canManageGym(resource.data.gymId);
      }

      match /gymStats/{gymId} {
        allow read: if verifiedEmail() || isSuperadmin();
        allow write: if isSuperadmin() || (verifiedEmail() && isSelf(uid));
      }

      match /progressHistory/{recordId} {
        allow read: if isSuperadmin() || (verifiedEmail() && isSelf(uid));
        allow create, update: if isSuperadmin() || (verifiedEmail() && isSelf(uid));
        allow delete: if isSuperadmin() || (verifiedEmail() && isSelf(uid));
      }

      match /climbEvents/{eventId} {
        allow read: if isSuperadmin() || (verifiedEmail() && isSelf(uid));
        allow create, update: if isSuperadmin() || (verifiedEmail() && isSelf(uid));
        allow delete: if isSuperadmin() || (verifiedEmail() && isSelf(uid));
      }
    }

    match /usernames/{usernameLower} {
      allow read: if true;
      allow create: if signedIn()
        && request.resource.data.uid == request.auth.uid
        && !exists(/databases/$(database)/documents/usernames/$(usernameLower))
        && request.resource.data.keys().hasOnly(["uid", "createdAt"]);
      allow update: if false;
      allow delete: if isSuperadmin() || (signedIn() && resource.data.uid == request.auth.uid);
    }

    match /follows/{followId} {
      allow read: if verifiedEmail() || isSuperadmin();
      allow create: if isSuperadmin() || (verifiedEmail() && request.resource.data.followerId == request.auth.uid);
      allow update, delete: if isSuperadmin() || (verifiedEmail() && resource.data.followerId == request.auth.uid);
    }

    match /gyms/{gymId} {
      allow read: if verifiedEmail() || isSuperadmin();
      allow create: if isSuperadmin() || (verifiedEmail() && request.resource.data.owner == request.auth.uid);
      allow update, delete: if canManageGym(gymId);

      match /Sectors/{sectorId} {
        allow read: if verifiedEmail() || isSuperadmin();
        allow write: if canManageGym(gymId);

        match /routes/{routeId} {
          allow read: if verifiedEmail() || isSuperadmin();
          allow write: if canManageGym(gymId);

          match /reviews/{reviewId} {
            allow read: if verifiedEmail() || isSuperadmin();
            allow create, update, delete: if canManageGym(gymId) || (verifiedEmail() && request.auth.uid == reviewId);
          }

          match /progressFeed/{entryId} {
            allow read: if verifiedEmail() || isSuperadmin();
            allow write: if canManageGym(gymId) || (verifiedEmail() && request.auth.uid == entryId);
          }
        }
      }

      match /models/{modelVersionId} {
        allow read: if verifiedEmail() || isSuperadmin();
        allow write: if canManageGym(gymId);
      }

      match /events/{eventId} {
        allow read: if verifiedEmail() || isSuperadmin();
        allow create, update, delete: if canManageGym(gymId);

        match /registrations/{userId} {
          allow list: if canManageGym(gymId);
          allow get: if canManageGym(gymId) || (verifiedEmail() && request.auth.uid == userId);
          allow create: if canManageGym(gymId) || (
            verifiedEmail()
            && request.auth.uid == userId
            && request.resource.data.userId == request.auth.uid
          );
          allow update: if canManageGym(gymId) || (
            verifiedEmail()
            && request.auth.uid == userId
            && resource.data.userId == request.auth.uid
            && request.resource.data.userId == request.auth.uid
          );
          allow delete: if canManageGym(gymId) || (
            verifiedEmail()
            && request.auth.uid == userId
            && resource.data.userId == request.auth.uid
          );
        }

        match /competitionLiveEntries/{userId} {
          function parentEventCompetitionLiveOpen() {
            return exists(/databases/$(database)/documents/gyms/$(gymId)/events/$(eventId))
              && get(/databases/$(database)/documents/gyms/$(gymId)/events/$(eventId)).data.competition_live.status != 'closed';
          }

          function hasValidCompetitionLiveEntryScore() {
            return request.resource.data.score == request.resource.data.completedRouteIds.size();
          }

          allow list: if canManageGym(gymId);
          allow get: if canManageGym(gymId) || (verifiedEmail() && request.auth.uid == userId);
          allow create: if (
            canManageGym(gymId) || (
              verifiedEmail()
              && request.auth.uid == userId
              && request.resource.data.userId == request.auth.uid
            )
          )
          && parentEventCompetitionLiveOpen()
          && hasValidCompetitionLiveEntryScore();
          allow update: if (
            canManageGym(gymId) || (
              verifiedEmail()
              && request.auth.uid == userId
              && resource.data.userId == request.auth.uid
              && request.resource.data.userId == request.auth.uid
            )
          )
          && parentEventCompetitionLiveOpen()
          && hasValidCompetitionLiveEntryScore();
          allow delete: if (
            canManageGym(gymId) || (
              verifiedEmail()
              && request.auth.uid == userId
              && resource.data.userId == request.auth.uid
            )
          )
          && parentEventCompetitionLiveOpen();
        }
      }

      match /members/{uid} {
        allow read: if verifiedEmail() || isSuperadmin() || canManageGym(gymId);
        allow create, update: if verifiedEmail()
          && request.auth.uid == uid
          && request.resource.data.uid == uid
          && request.resource.data.keys().hasOnly([
            "uid", "username", "firstName", "lastName", "city", "birthDate", "createdAt", "totalClimbsInGym", "lastActivityInGym", "status", "avatarUrl"
          ])
          && request.resource.data.status in ["active", "inactive"];
        allow delete: if false;
      }
    }

    match /socialActivities/{activityId} {
      allow read: if verifiedEmail() || isSuperadmin();
      allow create: if isSuperadmin() || verifiedEmail();
      allow update: if isSuperadmin() || verifiedEmail();
      allow delete: if isSuperadmin() || canManageGym(resource.data.gymId) || (verifiedEmail() && resource.data.user_id == request.auth.uid);

      match /applause/{applauseUid} {
        allow read: if verifiedEmail() || isSuperadmin();
        allow write: if isSuperadmin() || verifiedEmail();
      }

      match /comments/{commentId} {
        allow read: if verifiedEmail() || isSuperadmin();
        allow create: if isSuperadmin() || verifiedEmail();
        allow update, delete: if isSuperadmin() || canManageGym(get(/databases/$(database)/documents/socialActivities/$(activityId)).data.gymId) || verifiedEmail();
      }
    }

    match /gymOwnerRequests/{requestId} {
      allow create: if isSuperadmin() || (
        verifiedEmail()
        && request.resource.data.userId == request.auth.uid
        && request.resource.data.status == "pending"
        && request.resource.data.gymName is string
        && request.resource.data.gymName.size() > 0
        && request.resource.data.keys().hasOnly(["userId", "gymName", "status", "createdAt"])
      );

      allow read: if isSuperadmin() || (verifiedEmail() && resource.data.userId == request.auth.uid);
      allow update, delete: if isSuperadmin();
    }

    match /gymFollowers/{followId} {
      allow read: if verifiedEmail() || isSuperadmin();
      allow create: if isSuperadmin() || (
        verifiedEmail()
        && request.resource.data.userId == request.auth.uid
        && request.resource.data.gymId is string
        && request.resource.data.notificationsEnabled == true
      );
      allow update: if isSuperadmin() || (
        verifiedEmail()
        && resource.data.userId == request.auth.uid
        && request.resource.data.userId == resource.data.userId
      );
      allow delete: if isSuperadmin() || (verifiedEmail() && resource.data.userId == request.auth.uid);
    }

    match /notifications/{notificationId} {
      allow read: if isSuperadmin() || (verifiedEmail() && resource.data.userId == request.auth.uid);
      allow create: if isSuperadmin() || (
        verifiedEmail()
        && request.resource.data.read == false
        && request.resource.data.userId is string
        && request.resource.data.type is string
        && request.resource.data.message is string
        && (!request.resource.data.keys().hasAny(["actorId"]) || request.resource.data.actorId == null || request.resource.data.actorId is string)
        && (!request.resource.data.keys().hasAny(["postId"]) || request.resource.data.postId == null || request.resource.data.postId is string)
        && (!request.resource.data.keys().hasAny(["gymId"]) || request.resource.data.gymId == null || request.resource.data.gymId is string)
        && (!request.resource.data.keys().hasAny(["sectorId"]) || request.resource.data.sectorId == null || request.resource.data.sectorId is string)
        && request.resource.data.keys().hasOnly([
          "userId", "type", "message", "actorId", "postId", "gymId", "sectorId", "createdAt", "read"
        ])
        && (
          request.resource.data.userId == request.auth.uid
          || (
            request.resource.data.gymId is string
            && canManageGym(request.resource.data.gymId)
          )
          || (
            request.resource.data.type in ["POST_LIKE", "POST_COMMENT"]
            && request.resource.data.actorId is string
            && request.resource.data.actorId == request.auth.uid
            && request.resource.data.postId is string
            && exists(/databases/$(database)/documents/socialActivities/$(request.resource.data.postId))
            && (
              get(/databases/$(database)/documents/socialActivities/$(request.resource.data.postId)).data.user_id == request.resource.data.userId
              || get(/databases/$(database)/documents/socialActivities/$(request.resource.data.postId)).data.userId == request.resource.data.userId
            )
          )
        )
      );
      allow update: if isSuperadmin() || (
        verifiedEmail()
        && resource.data.userId == request.auth.uid
        && request.resource.data.userId == resource.data.userId
      );
      allow delete: if isSuperadmin() || (verifiedEmail() && resource.data.userId == request.auth.uid);
    }

    match /analytics/superadmin_kpis {
      allow read: if isSuperadmin();
      allow write: if false;
    }

    match /analytics/{docId} {
      allow read: if isSuperadmin();
      allow write: if false;
    }

    match /{document=**} {
      allow read: if isSuperadmin();
      allow write: if false;
    }
  }
}
