rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {

    function signedIn() {
      return request.auth != null;
    }

    function verifiedEmail() {
      return signedIn() && request.auth.token.email_verified == true;
    }

    function myUid() {
      return request.auth.uid;
    }

    function userDoc() {
      return firestore.get(/databases/(default)/documents/users/$(myUid()));
    }

    function role() {
      return signedIn() &&
        firestore.exists(/databases/(default)/documents/users/$(myUid()))
        ? userDoc().data.role
        : null;
    }

    function isSuperadmin() {
      return role() == 'superadmin';
    }

    function isGymAdminUser() {
      return role() == 'gym_admin' || role() == 'admin';
    }

    function isGymOwner(gymId) {
      return signedIn()
        && firestore.exists(/databases/(default)/documents/gyms/$(gymId))
        && firestore.get(/databases/(default)/documents/gyms/$(gymId)).data.owner == myUid();
    }

    function managesGym(gymId) {
      return signedIn()
        && firestore.exists(/databases/(default)/documents/users/$(myUid()))
        && isGymAdminUser()
        && userDoc().data.gymManaged == gymId;
    }

    function canManageGym(gymId) {
      return isSuperadmin() || isGymOwner(gymId) || managesGym(gymId);
    }

    // 🔵 3D MESHES (PRODUZIONE)
    match /meshes/{gymId}/{fileName} {

      // lettura: utenti autenticati e verificati
      allow read: if verifiedEmail() || isSuperadmin();

      // scrittura: solo admin del gym + solo file .glb
      allow write: if fileName.matches('.*\\.glb$')
        && canManageGym(gymId);
    }

    // 🟣 FOTO SETTORE
    match /sector-media/{gymId}/{sectorId}/{fileName} {

      // lettura: utenti autenticati e verificati
      allow read: if verifiedEmail() || isSuperadmin();

      // scrittura: solo admin del gym + solo file immagine
      allow write: if canManageGym(gymId)
        && request.resource.size < 8 * 1024 * 1024
        && request.resource.contentType.matches('image/.*')
        && fileName.matches('photo\\.[A-Za-z0-9]+$');
    }

    // 🟢 MEDIA PALESTRA (logo + piantina)
    match /gymMedia/{gymId}/{fileName} {
      allow read: if verifiedEmail() || isSuperadmin();

      allow create, update: if verifiedEmail()
        && canManageGym(gymId)
        && request.resource.size < 8 * 1024 * 1024
        && request.resource.contentType.matches('image/.*');

      allow delete: if verifiedEmail()
        && canManageGym(gymId);
    }

    // 🟡 AVATAR UTENTE
    match /avatars/{uid}/{fileName} {
      allow read: if verifiedEmail() || isSuperadmin();

      allow create, update: if verifiedEmail()
        && uid == myUid()
        && request.resource.size < 5 * 1024 * 1024
        && request.resource.contentType.matches('image/.*');

      allow delete: if verifiedEmail()
        && uid == myUid();
    }

    // 🔴 BLOCCA TUTTO IL RESTO
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
